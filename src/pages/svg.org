* path
- M :: moveto
- L :: lineto
- H x :: horizontal lineto
- h dx :: 相対版
- V y :: vertical lineto
- v dy :: 相対版
- C x1 y1, x2 y2, x y :: curveto 3次のbezier curve
- S x2 y2, x y :: 3次のbezier curveだが前の制御点の反対を付けてくれる
- s dx2 dy2, dx dy :: 相対版
- Z :: closepath
